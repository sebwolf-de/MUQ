167 void RootfindingIVP::JacobianImpl(unsigned int const wrtIn, unsigned int const wrtOut, ref_vector<boost::any> const& inputs) {
168  if( wrtOut>0 ) {
169  std::cerr << std::endl << "ERROR: Cannot compute derivative infomration of muq::Modeling::RootfindingIVP for outputs other than the state at the root (the first output)" << std::endl << std::endl;
170 
171  assert(wrtOut==0);
172  }
173 
174  // find the first root to one of the root outputs
175  const Eigen::VectorXi& rootsfound = FindRoot(inputs, wrtIn, wrtOut);
176  unsigned int ind = 0;
177  for( ; ind<rootsfound.size(); ++ind ) {
178  if( rootsfound(ind)!=0 ) { break; }
179  }
180 
181  // the inputs to the rhs function
182  ref_vector<boost::any> rhsIns;
183  rhsIns.push_back(outputs[0]);
184  rhsIns.insert(rhsIns.end(), inputs.begin()+1, inputs.begin()+rhs->numInputs);
185 
186  // the inputs to the root function
187  ref_vector<boost::any> rootIns;
188  rootIns.push_back(outputs[0]);
189  rootIns.insert(rootIns.end(), inputs.begin()+rhs->numInputs, inputs.begin()+rhs->numInputs+root->numInputs-1);
190 
191  // the derivative of the root function at the final time -- derivative of the root wrt the state
192  const boost::any& dgdy = root->Jacobian(0, ind, rootIns);
193  const DlsMat& dgdyref = boost::any_cast<const DlsMat&>(dgdy); // 1 x stateSize matrix
194 
195  // evaluate the right hand side
196  const std::vector<boost::any>& f = rhs->Evaluate(rhsIns); // stateSize x 1 vector
197  const N_Vector& fref = boost::any_cast<N_Vector>(f[0]);
198  assert(NV_LENGTH_S(fref)==dgdyref->N);
199 
200  double dum = 0.0;
201  for( unsigned int i=0; i<dgdyref->N; ++i ) {
202  dum += DENSE_ELEM(dgdyref, 0, i)*NV_Ith_S(fref, i);
203  }
204 
205  if( wrtIn>=rhs->numInputs ) { // derivative wrt root parameter
206  boost::any dgdpara = root->Jacobian(wrtIn-rhs->numInputs+1, ind, rootIns);
207  const DlsMat& dgdpararef = boost::any_cast<const DlsMat&>(dgdpara); // 1 x paramSize matrix
208  DenseScale(-1.0/dum, dgdpararef);
209 
210  jacobian = NewDenseMat(NV_LENGTH_S(fref), dgdpararef->N);
211  DlsMat& jac = boost::any_cast<DlsMat&>(*jacobian); // stateSize x paramSize matrix
212  SetToZero(jac);
213 
214  // update the jacobian jac += f.transpose()*dgdpara (outer product)
215  for( unsigned int i=0; i<jac->M; ++i ) {
216  for( unsigned int j=0; j<jac->N; ++j ) {
217  DENSE_ELEM(jac, i, j) = NV_Ith_S(fref, i)*DENSE_ELEM(dgdpararef, 0, j);
218  }
219  }
220 
221  return;
222  }
223 
224  // get a reference to the jacobian
225  DlsMat& jac = boost::any_cast<DlsMat&>(*jacobian); // stateSize x paramSize matrix
226  assert(jac->M==dgdyref->N);
227  assert(NV_LENGTH_S(fref)==jac->M);
228 
229  if( wrtIn==0 ) { // derivative wrt the initial conditions
230  // add the identity
231  AddIdentity(jac);
232  }
233 
234  DenseScale(-1.0/dum, dgdyref);
235 
236  // the derivative of the final time wrt to the parameter; dtfdpara = -jac_{state}(root)*jac/dot(jac_{state}(root), f)
237  N_Vector dtfdic = N_VNew_Serial(jac->N);
238  for( unsigned int i=0; i<jac->N; ++i ) {
239  NV_Ith_S(dtfdic, i) = 0.0;
240  for( unsigned int j=0; j<jac->M; ++j ) {
241  NV_Ith_S(dtfdic, i) += DENSE_ELEM(dgdyref, 0, j)*DENSE_ELEM(jac, j, i);
242  }
243  }
244 
245  // update the jacobian jac += f.transpose()*dtfdpara (outer product)
246  for( unsigned int i=0; i<jac->M; ++i ) {
247  for( unsigned int j=0; j<jac->N; ++j ) {
248  DENSE_ELEM(jac, i, j) += NV_Ith_S(fref, i)*NV_Ith_S(dtfdic, j);
249  }
250  }
251 
252  // destroy temp vectors/matrices
253  N_VDestroy(dtfdic);
254 }
